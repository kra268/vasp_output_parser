def get_name(filename):
    """As a standard, the code follows a certain nomenclature for the directories.
    This function simply splits the file name into three components"""

    components = filename.split('_')
    return components[0], components[1], components[2]


def make_dirs(files):
    """A separate directory is made for each OUTCAR file in the OUTCAR_files directory"""

    for file in files:
        file_type, molecule, site = get_name(file)
        # print(file_type, molecule, site)
        if file_type == 'OUTCAR':
            poscar_dir = os.getcwd() + '/POSCAR_files/' + f'POSCAR_{molecule}_{site}'
            # print(poscar_dir)
            if not os.path.exists(poscar_dir):
                os.makedirs(poscar_dir)
                print(f'Made a directory called POSCAR_{molecule}_{site}')
            else:
                print(f'POSCAR_{molecule}_{site} already exists')
